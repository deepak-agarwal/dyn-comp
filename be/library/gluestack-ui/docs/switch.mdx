### usage

```jsx
<Switch />
```

### examples

```jsx
import { Switch } from '@gluestack-ui/themed'

const switchExample = () => {
  const [checked, setChecked] = React.useState(false)

  return (
    <>
      <Switch
        checked={checked}
        onChange={() => setChecked(!checked)}
        label='Switch'
      />
      <Switch
        checked={checked}
        onChange={() => setChecked(!checked)}
        label='Switch'
        disabled
      />
    </>
  )
}
```
