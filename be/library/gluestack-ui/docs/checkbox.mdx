```jsx
  CheckboxIcon,
  CheckboxLabel
} from '@gluestack-ui/themed'
```

### Usage

```jsx
<CheckboxGroup>
  <Checkbox>
    <CheckboxIndicator>
      <CheckboxIcon />
    </CheckboxIndicator>
    <CheckboxLabel />
  </Checkbox>
</CheckboxGroup>
```

### Examples

#### Checkbox

```jsx
import {
  Checkbox,
  CheckboxIndicator,
  CheckboxIcon,
  CheckboxLabel
} from '@gluestack-ui/themed'

const CheckBoxExample = (props) => {
  return (
    <Checkbox {...props}>
      <CheckboxIndicator mr='$2'>
        <CheckboxIcon as={CheckIcon} />
      </CheckboxIndicator>
      <CheckboxLabel>Label</CheckboxLabel>
    </Checkbox>
  )
}
```

```jsx
import { useState } from 'react'
import {
  Checkbox,
  CheckboxGroup,
  CheckboxIndicator,
  CheckboxIcon,
  CheckboxLabel,
  Center,
  CheckIcon,
  Icon,
  VStack
} from '@gluestack-ui/themed'

const CheckBoxExample = (props) => {
  const [values, setValues] = useState(['Eng'])
  return (
    <CheckboxGroup
      value={values}
      onChange={(keys) => {
        setValues(keys)
      }}
    >
      <VStack space='3xl'>
        <Checkbox value='Eng'>
          <CheckboxIndicator mr='$2'>
            <CheckboxIcon as={CheckIcon} />
          </CheckboxIndicator>
          <CheckboxLabel>Framer</CheckboxLabel>
        </Checkbox>
        <Checkbox value='invison'>
          <CheckboxIndicator mr='$2'>
            <CheckboxIcon as={CheckIcon} />
          </CheckboxIndicator>
          <CheckboxLabel>Invision Studio</CheckboxLabel>
        </Checkbox>
        <Checkbox value='adobe'>
          <CheckboxIndicator mr='$2'>
            <CheckboxIcon as={CheckIcon} />
          </CheckboxIndicator>
          <CheckboxLabel>Adobe XD</CheckboxLabel>
        </Checkbox>
      </VStack>
    </CheckboxGroup>
  )
}
```
